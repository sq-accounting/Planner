📁 Google Drive (Apps Script)
📝 3 минутта — Drive-ға қосу:
1. script.google.com-ға кіріңіз (өз Google аккаунтыңызбен)
2. «New project» → бар кодты өшіріп, мынаны қойыңыз:
const FILE_NAME = 'planner_sync.json';
const SECRET = 'mysecret_2026';  // ← ӨЗГЕРТІҢІЗ!

function doGet(e){
  if(e.parameter.secret!==SECRET) return _j({error:'auth'});
  if(e.parameter.action==='ping') return _j({ok:true});
  const f=DriveApp.getFilesByName(FILE_NAME);
  if(!f.hasNext()) return _j({keys:{}});
  return ContentService.createTextOutput(
    f.next().getBlob().getDataAsString()
  ).setMimeType(ContentService.MimeType.JSON);
}
function doPost(e){
  if(e.parameter.secret!==SECRET) return _j({error:'auth'});
  const d=e.postData.contents;
  try{JSON.parse(d);}catch(_){return _j({error:'bad json'});}
  const f=DriveApp.getFilesByName(FILE_NAME);
  if(f.hasNext()) f.next().setContent(d);
  else DriveApp.createFile(FILE_NAME, d, MimeType.PLAIN_TEXT);
  return _j({ok:true});
}
function _j(o){return ContentService.createTextOutput(JSON.stringify(o))
  .setMimeType(ContentService.MimeType.JSON);}
